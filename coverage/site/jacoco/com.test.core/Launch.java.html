package com.test.core;

public class Launch {

	public static void main(String[] args) {
		WebDriver driver;
		 
		  DesiredCapabilities dCaps;
		 dCaps = new DesiredCapabilities();
		  dCaps.setJavascriptEnabled(true);
		  dCaps.setCapability("takesScreenshot", false);
		  dCaps.setCapability("phantomjs.binary.path", "C:\\Users\\450405\\.jenkins\\workspace\\TestAutomation\\resources\\phantomjs-2.1.1-windows\\bin\\phantomjs.exe");
		  driver = new PhantomJSDriver(dCaps);
		  driver.get("http://assertselenium.com/");
	}

}
